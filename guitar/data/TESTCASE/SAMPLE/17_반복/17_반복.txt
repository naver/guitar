;-----------------------------------------------------------------------------------------------------------------------------------------
;# 지정된 횟수만큼 이중 구조로 반복문을 수행
;-----------------------------------------------------------------------------------------------------------------------------------------


"$i=1" 설정 하고, "$i=3" 까지 반복한다. {

	$i 초 대기한다.

	"$j=1" 설정 하고, "$j=2" 까지 반복한다. {
		0.1초 대기한다.
		0.2초 대기한다.
	}

	$i=3 같으면 {
		0.7초 대기한다.
	}
}
